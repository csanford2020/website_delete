converting data types- explicit coercion

Number vs Character

x <- 1:3
typeof(x)
## [1] "integer"
typeof(as.character(x))
## [1] "character"

number vs logical

x <- c(TRUE, FALSE)

## [1]  TRUE FALSE
typeof(x)
## [1] "logical"
y <- as.numeric(x)

## [1] 1 0
